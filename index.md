  function regimentChangeHandler () {
    var regimentEl = document.getElementById('regiment')

    cost = getMoney(cost);
    document.getElementById('cost').value = cost ? 'R$ ' + formatReal(cost) : 'R$ 0,00';
    document.getElementById('step3').style.display = 'block';

  function getMoney( str ) {
    return parseInt( str.replace(/[\D]+/g,'') );
  }

  function formatReal( value ) {
    var tmp = value + '';
    if (tmp.length == 1) return '0,0' + tmp;
    if (tmp.length == 2) return '0,' + tmp;
    tmp = tmp.replace(/([0-9]{2})$/g, ",$1");
    if( tmp.length > 6)
      tmp = tmp.replace(/([0-9]{3}),([0-9]{2}$)/g, ".$1,$2");
    return tmp;
  }

  function money2float (value) {
    var tmp = value.replace('R$ ', '');
    tmp = tmp.replace(/\./g, '');
    tmp = tmp.replace(/\,/g, '.');
    return parseFloat(tmp)
  }

  function float2money (value) {
    value += ''
    var int = value.split('.')[0];
    var decimal = value.split('.')[1];
    decimal = (decimal ? decimal.substring(0,2) : '00');
    if (decimal.length === 1) decimal = decimal + '0';
    var li = int.length;
    var sm = parseInt(li / 3);
    var regx = "", repl = "";

    for (var i = 0; i < sm; i++) {
        regx = regx.concat('([0-9]{3})');
        repl = repl.concat('.$' + (i + 1));
    }
    regx = regx.concat('$');
    int = int.replace(new RegExp(regx, 'g'), repl);
    int = (li % 3) === 0 ? int.substr(1) : int;
    if (int.length === 1) int = '0' + int;
    int = 'R$ ' + int + ',' + decimal;
    return int;
  }

  function simulate() {
    var regimentEl = document.getElementById('regiment')
    var regiment = regimentEl.options[regimentEl.selectedIndex].value;
    var reg = {};
    var monthsEl = document.getElementById('months')
    var months = monthsEl.options[monthsEl.selectedIndex].value;
    var cost = document.getElementById('cost').value;
    var employees = document.getElementById('employees').value;
    cost = money2float(cost);
    switch (regiment) {
      case '1':
        reg = { val: 49.90, text: 'R$ 49,90', anualVal: 598.8, anualText: 'R$ 598,80'};
        employees = 0;
        break;
      case '2':
        var val = 85.90, text = 'R$ 85,90', anualVal = 1030.8, anualText = 'R$ 1.030,80';

        if (employees > 3) {
          val += (employees - 3) * 25,00;
          anualVal = val * 12;
          text = float2money(val);
          anualText = float2money(anualVal);
        }

        reg = { val: val, text: text, anualVal: anualVal, anualText: anualText};
        monthsEl = document.getElementById('monthsSimple')
        months = monthsEl.options[monthsEl.selectedIndex].value;
        break;
      case '3':
        var val = 199.90, text = 'R$ 199,90', anualVal = 2398.8, anualText = 'R$ 2.398,80';

        if (employees > 3) {
          val += (employees - 3) * 25,00;
          anualVal = val * 12;
          text = float2money(val);
          anualText = float2money(anualVal);
        }

        reg = { val: val, text: text, anualVal: anualVal, anualText: anualText};
        monthsEl = document.getElementById('monthsSimple')
        months = monthsEl.options[monthsEl.selectedIndex].value;
        break;
    }

    // gasto com contabilidade anual
    var spend = Math.round((cost * months) * 100) / 100;
    spendText = float2money(spend);
    document.getElementById('spend').innerHTML = 'Atualmente você gasta <b> '+ spendText +'</b> por ano em contabilidade.'

    // valor do plano
    var int = reg.text.split(',')[0];
    var dec = reg.text.split(',')[1];
    document.getElementById('monval').innerHTML = int +',<span class="cents">'+ dec +'</span> <span class="time">/ mês</span>';

    // valor do plano anual
    int = reg.anualText.split(',')[0];
    dec = reg.anualText.split(',')[1];
    document.getElementById('anualval').innerHTML = int +',<span class="cents">'+ dec +'</span> <span class="time">/ ano</span>';

    // economia
    var economy = spend - reg.anualVal;
    economy = economy < 0 ? 0 : economy;
    var economyText = float2money(economy);
    int = economyText.split(',')[0];
    dec = economyText.split(',')[1];
    document.getElementById('economy').innerHTML = int +',<span class="cents">'+ dec +'</span> <span class="time">/ ano</span>';
  }

  function showResult() {
    document.getElementById('resultado').style.display = 'block';
  }

  function simulateAndShowResult() {
    simulate();
    showResult();
  }
